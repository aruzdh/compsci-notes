\section{Lenguajes Regulares}

\dfn{Lenguajes Regulares}{
  Un \textbf{lenguajes regular} se define recursivamente de la siguiente manera.
  \begin{enumerate}[a)]
    \item $\varnothing$ y $\{\varepsilon\}$ son lenguajes regulares.
    \item Si $a \in \Sigma$, entonces $\{a\}$ es un lenguaje regular.
    \item Si \textit{L} y \textit{M} son lenguajes regulares, entonces $L \cup M,~ L^*,~ LM$ son lenguajes regulares (considerese los casos análogos).
    \item Solo los anteriores son lenguajes regulares.
  \end{enumerate}
}

\ex{Lenguajes Regulares}{
    ¿ Los lenguajes finitos son regulares ?\\
    Consideremos
    \[L = \{w_1, w_2, \cdots, w_i\}\] donde cada $w_i = a_{i1}a_{i2}\cdots a_{im}$ es un lenguaje regular por \textit{b)}, ya que por \textit{a)} se tiene que $\{a_{ij}\}$ es un lenguaje regular.\\
    Por tanto
    \[L = \bigcup_{i = 1}^k w_i\] es un lenguaje regular.
}

\ex{Lenguajes Regulares}{
  ¿ Un alfabeto es un lenguaje regular ?\\
  Consideremos $\Sigma = \{a_1, a_2, \cdots ,a_n\}$. Se tiene que $\{a_i\}$ es un lenguaje regular.\\
  Por tanto
  \[\bigcup_{i = 1}^n a_i\] es un lenguaje regular.
}

\ex{Lenguajes Regulares}{
 ¿ $\Sigma^*$ es un lenguaje regular ? \\
 Se tiene que $\Sigma$ es un lenguaje regular. Se sigue de la definición que $\Sigma^*$ en un lenguaje regular.
}

\ex{Lenguajes Regulares}{
  Sea $\Sigma = \{a,b\}$. \textit{L} es el lenguaje de todas las cadenas de $\Sigma^*$ que tienen como prefijo la cadena \textit{bb}. \\
  ¿L es un lenguaje regular ?\\
  Se tiene que $\{b\}$ es un lenguaje regular, y por tanto $\{b\}\{b\}$ también lo es. Además, $\Sigma^* = \{a,b\}^*$ es regular.\\
  Por tanto \[\{b\}\{b\}\{a,b\}^* = L\] es un lenguaje regular.
}

\ex{Lenguajes Regulares}{
  Sea $\Sigma = \{a,b\}$. \textit{L} es el lenguaje que contiene cadenas con \textit{exactamente} una \textit{b}.\\
  ¿L es un lenguaje regular ?\\
  Se tiene que $\{a\}^*, \{b\}$ son lenguajes regulares (se sigue de los ejemplos pasados).\\
  Por tanto \[\{a\}^*\{b\}\{a\}^* = L\] es un lenguaje regular.
}

\ex{Lenguajes Regulares}{
  Sea \textit{L} el lenguaje de las cadenas que contiene un número par de \textit{b's}. L es un lenguaje regular debido a que
  \[L = (\{a\}^* \{b\} \{a\}^* \{b\} \{a\}^*)^* \cup \{a\}^*\]
}
