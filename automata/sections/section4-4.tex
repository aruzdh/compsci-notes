\section{Gramáticas Libres de Contexto}
\dfn{Gramática Libre de Contexto}{
  Una \textbf{gramática libre de contexto} \textit{G} cumple que sus producciones son de la forma
  \[A \rightarrow w\]
  con $A \in V,~ w \in (V \cup T)^*$
}

\ex{Gramática Libre de Contexto}{
  Se desea generar el lenguaje $L = \{a^nb^n \mid n \ge 0\}$. Una gramática para lo anterior se mira de la siguiente manera.
  \[S \rightarrow aSb \mid \varepsilon\]
}

\ex{Gramática Libre de Contexto}{
  Se desea generar el lenguaje $L = \{a^nb^ma^mb^ \mid n,m \ge 0\}$. Una gramática para lo anterior se mira de la siguiente manera.
  \begin{align*}
    &S \rightarrow aSb \mid bAa \mid \varepsilon\\
    &A \rightarrow bAa \mid \varepsilon
  \end{align*}
}

\nt{
  $a^nb^ma^nb^m$ y $a^nb^nc^n$ \textbf{no} son libre de contexto.
}

\ex{Gramática Libre de Contexto}{
  Se diseña una gramática qu genere $L = \{ww^R \mid w \in \{0,1\}^*\}$.
  \[S \rightarrow \varepsilon \mid 0S0 \mid 1S1 \]
}

\ex{Gramática Libre de Contexto}{
  Se diseña una gramática qu genere $L = \{w \mid w \text{ es palindromo }\}$.
  \[S \rightarrow \varepsilon \mid 0 \mid 1 \mid 0S0 \mid 1S1 \]
}

\ex{Gramática Libre de Contexto}{
  Las siguiente son opciones diferentes para generar $L = \{a^nb^m \mid n \ge m\}$.
  \begin{itemize}
    \item Opción 1
      \begin{align*}
        &S \rightarrow \varepsilon \mid AaBb \mid aBb \mid A\\
        &A \rightarrow a \mid aA\\
        &B \rightarrow aBb \mid \varepsilon
      \end{align*}

    \item Opción 2
      \begin{align*}
        &S \rightarrow \varepsilon \mid aBb\\
        &B \rightarrow a \mid aB \mid aBb \varepsilon \mid a
      \end{align*}

    \item Opción 3
      \begin{align*}
        &S \rightarrow \varepsilon \mid aSb \mid aS
      \end{align*}

  \end{itemize}
}
